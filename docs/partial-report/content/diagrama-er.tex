\tikzset{
  erd entity/.style={
    rectangle, draw=#1, line width=0.7pt, fill=#1!4,
    inner sep=0pt, outer sep=0pt, minimum width=3.4cm,
  },
  erd head/.style={
    rectangle, fill=#1, text=white, font=\bfseries\scriptsize,
    minimum width=3.4cm, minimum height=0.45cm, inner sep=2pt, anchor=north,
  },
  erd fields/.style={
    rectangle, draw=#1, fill=#1!4, line width=0.7pt,
    minimum width=3.4cm, inner sep=3pt, anchor=north,
    font=\tiny, text width=3.1cm, align=left,
  },
  fk arrow/.style={-{Stealth[length=4pt, width=3pt]}, semithick, #1!70!black},
  m2m arrow/.style={{Stealth[length=4pt, width=3pt]}-{Stealth[length=4pt, width=3pt]}, semithick, #1!70!black, dashed},
  mod label/.style={font=\scriptsize\bfseries\itshape, text=#1},
}

\newcommand{\ent}[5]{% name, color, title, fields, (x,y)
  \node[erd head=#2] (#1H) at (#5) {#3};
  \node[erd fields=#2, below=-0.7pt of #1H] (#1B) {#4};
  \node[fit=(#1H)(#1B), inner sep=0pt] (#1) {};
}
