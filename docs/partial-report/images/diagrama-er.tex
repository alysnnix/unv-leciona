\tikzset{
    entity header/.style={
        rectangle,
        fill=#1,
        text=white,
        font=\bfseries\footnotesize,
        minimum width=3.6cm,
        minimum height=0.55cm,
        inner sep=2pt,
        anchor=north,
    },
    entity body/.style={
        rectangle,
        fill=#1!5,
        draw=#1,
        line width=0.8pt,
        minimum width=3.6cm,
        inner sep=3pt,
        font=\scriptsize,
        text width=3.2cm,
        anchor=north,
    },
    rel/.style={
        -{Stealth[length=5pt]},
        semithick,
        #1,
    },
    m2m/.style={
        {Stealth[length=5pt]}-{Stealth[length=5pt]},
        semithick,
        #1,
        dashed,
    },
    module label/.style={
        font=\scriptsize\bfseries\itshape,
        text=#1,
    },
}

\newcommand{\entityNode}[5]{%
    \node[entity header=#2] (#1-header) at (#5) {#3};
    \node[entity body=#2, below=0pt of #1-header] (#1-body) {#4};
    \coordinate (#1) at ($(#1-header.north)!0.5!(#1-body.south)$);
    \node[fit=(#1-header)(#1-body), inner sep=0pt] (#1-box) {};
}
